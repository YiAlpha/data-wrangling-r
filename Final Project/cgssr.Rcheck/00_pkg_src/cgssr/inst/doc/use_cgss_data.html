Search question to obtian the index in the dataset

library(cgssr)
data(cgss2003)
data(codebook2003_en)
ques_search_en("trust",codebook2003_en)
#>        index                  question_en
#> 607 truststr Degree of trust in strangers

Get to know the answer labels of the variable

codebook2003_en$label_en[codebook2003_en$index=='truststr']
#> [1] "c (not suitable = -3, `no answer / no answer ` = -2, `do not know / can't tell ` = -1, very untrusted = 1, no trust = 2, general = 3, trust = 4 , very trust = 5)"
